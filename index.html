function doGet(e) { try { const action = e.parameter.action || ''; if (action === 'suggestions') { const type = (e.parameter.type || '').toLowerCase(); const suggestions = getSuggestions_(type); return jsonOutput_({ success: true, suggestions: suggestions }); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } return sheet; } function ensureHeaders_(sheet) { const lastColumn = sheet.getLastColumn(); const existingHeaders = lastColumn > 0 ? sheet.getRange(1, 1, 1, lastColumn).getValues()[0] : []; const isEmpty = existingHeaders.join('').trim() === ''; if (isEmpty) { sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); sheet.setFrozenRows(1); return; } const existingJoined = existingHeaders.slice(0, HEADERS.length).join('|'); const requiredJoined = HEADERS.join('|'); if (existingJoined !== requiredJoined) { sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); sheet.setFrozenRows(1); } } function getSuggestions_(type) { const sheet = getOrCreateSheet_(); } else if (type === 'contractor') { columnName = 'Contractor Office'; } else { return []; } const colIndex = headers.indexOf(columnName); if (colIndex === -1) return []; const values = data .slice(1) .map(row => row[colIndex]) .filter(v => v !== null && v !== undefined && String(v).trim() !== '') .map(v => String(v).trim()); const unique = [...new Set(values)]; unique.sort((a, b) => a.localeCompare(b, 'ar')); return unique.slice(0, 200); } function valueOrEmpty_(value) { return value === undefined || value === null ?
